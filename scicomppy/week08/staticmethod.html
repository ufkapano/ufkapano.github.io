@staticmethod decorator

https://docs.python.org/3/library/functions.html

INTRODUCTION

A static method does not receive an implicit first argument (self). 

# Syntax.

class C:
    @staticmethod
    def f(arg1, arg2, ...): ...

# Calling: C.f() or C().f()

POLYNOMIALS

class Poly:

    def __init__(self, c=0, n=0):
        self.data = (n+1) * [0]
        self.data[-1] = c

    @staticmethod
    def gcd(a, b):
        if a % b == 0:
            return b
        else:
            return Poly.gcd(b, a % b)

assert Poly.gcd(3, 5) == 1
assert Poly.gcd(4, 6) == 2
